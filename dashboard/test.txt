// ===================== CONVERT WATER CM TO PERCENTAGE =====================
function waterCmToPercent(waterCm) {
  const waterLevelCm = maxTankLevelCm - waterCm;
  if (waterLevelCm < 0) return 0;
  if (waterLevelCm > maxTankLevelCm) return 100;
  return (waterLevelCm / maxTankLevelCm) * 100;
}

function waterCmToLiters(waterCm) {
  const waterLevelCm = maxTankLevelCm - waterCm;
  if (waterLevelCm < 0) return 0;
  if (waterLevelCm > maxTankLevelCm) return maxCapacityLiters;
  return (waterLevelCm / maxTankLevelCm) * maxCapacityLiters;
}

function getWaterHeightCm(waterCm) {
  const waterLevelCm = maxTankLevelCm - waterCm;
  if (waterLevelCm < 0) return 0;
  if (waterLevelCm > maxTankLevelCm) return maxTankLevelCm;
  return waterLevelCm;
}

function calculateWaterNeed(d) {
    let needScore = 0;
    
    if (d.soil < 20) {
        needScore += 60;
    } else if (d.soil < 30) {
        needScore += 50;
    } else if (d.soil < 40) {
        needScore += 35;
    } else if (d.soil < 50) {
        needScore += 20;
    } else if (d.soil > 70) {
        needScore -= 15;
    }
    
    if (d.temp > 38) {
        needScore += 25;
    } else if (d.temp > 33) {
        needScore += 18;
    } else if (d.temp > 28) {
        needScore += 10;
    } else if (d.temp < 15) {
        needScore -= 10;
    }
    
    if (d.hum < 25) {
        needScore += 15;
    } else if (d.hum < 35) {
        needScore += 10;
    } else if (d.hum > 75) {
        needScore -= 10;
    }
    
    const waterPercent = waterCmToPercent(d.water);
    if (waterPercent < 15) {
        needScore = 0;
    } else if (waterPercent < 30) {
        needScore *= 0.5;
    }
    
    if (weatherRainExpected) {
        needScore *= 0.6;
    }
    
    return Math.min(100, Math.max(0, Math.round(needScore)));
}

function runAI(d) {
    if (!isAIActive) return;
    
    const waterNeed = calculateWaterNeed(d);
    const waterPercent = waterCmToPercent(d.water);
    const tankHasWater = waterPercent > 15;
    
    let shouldRun = false;
    
    if (waterNeed > 50 && tankHasWater && !weatherRainExpected) {
        shouldRun = true;
    } else if (waterNeed > 70 && tankHasWater) {
        shouldRun = true;
    } else if (waterNeed < 25) {
        shouldRun = false;
    } else if (!tankHasWater) {
        shouldRun = false;
    } else if (weatherRainExpected && waterNeed < 55) {
        shouldRun = false;
    }
    
    if (shouldRun !== lastPumpState) {
        pumpRef.set(shouldRun ? 1 : 0);
        lastPumpState = shouldRun;
        console.log(`🤖 AI: ${shouldRun ? "STARTED" : "STOPPED"} pump - Need: ${waterNeed}%`);
    }
    
    const insightText = document.getElementById("smartInsight");
    if (insightText) {
        if (shouldRun) {
            insightText.innerHTML = `🤖 AI: Watering (Need: ${waterNeed}%)`;
        } else {
            let reason = "Conditions optimal";
            if (waterNeed < 25) reason = "Soil moisture sufficient";
            if (!tankHasWater) reason = "Low water tank";
            if (weatherRainExpected) reason = "Rain expected";
            insightText.innerHTML = `🤖 AI: Idle - ${reason}`;
        }
    }
}

// ================= HISTORY LISTENER =================
db.ref("history").on("value", snap => {
    historyData = [];
    const hourMap = new Map();
    
    snap.forEach(child => {
        const v = child.val();
        if(!v) return;
        
        const waterPercent = waterCmToPercent(v.water);
        const date = new Date(v.time);
        const hourKey = `${date.toDateString()}_${date.getHours()}`;
        
        if (!hourMap.has(hourKey) || v.time > hourMap.get(hourKey).time) {
            hourMap.set(hourKey, {
                temp: +v.temp || 0,
                hum: +v.hum || 0,
                soil: +v.soil || 0,
                water: waterPercent,
                waterRaw: v.water,
                time: +v.time || Date.now(),
                hour: date.getHours(),
                date: date
            });
        }
    });
    
    historyData = Array.from(hourMap.values());
    historyData.sort((a, b) => a.time - b.time);
    buildChart();
});

function buildChart() {
    chart.data.labels = [];
    chart.data.datasets.forEach(ds => ds.data = []);
    
    if (historyData.length === 0) {
        chart.update();
        return;
    }
    
    let displayData = [];
    
    if (mode === "hour") {
        displayData = historyData.slice(-12);
        
        displayData.forEach(d => {
            const date = new Date(d.time);
            let hour = date.getHours();
            let ampm = hour >= 12 ? 'PM' : 'AM';
            let hour12 = hour % 12;
            hour12 = hour12 ? hour12 : 12;
            chart.data.labels.push(`${hour12}:00 ${ampm}`);
            chart.data.datasets[0].data.push(d.temp);
            chart.data.datasets[1].data.push(d.hum);
            chart.data.datasets[2].data.push(d.soil);
            chart.data.datasets[3].data.push(d.water);
        });
    } 
    else if (mode === "day") {
        const dailyMap = new Map();
        
        historyData.forEach(d => {
            const date = new Date(d.time);
            const dayKey = date.toLocaleDateString("en-GB");
            if (!dailyMap.has(dayKey) || d.time > dailyMap.get(dayKey).time) {
                dailyMap.set(dayKey, {
                    temp: d.temp,
                    hum: d.hum,
                    soil: d.soil,
                    water: d.water,
                    time: d.time,
                    label: dayKey
                });
            }
        });
        
        const last7Days = Array.from(dailyMap.values()).slice(-7);
        last7Days.forEach(day => {
            chart.data.labels.push(day.label);
            chart.data.datasets[0].data.push(day.temp);
            chart.data.datasets[1].data.push(day.hum);
            chart.data.datasets[2].data.push(day.soil);
            chart.data.datasets[3].data.push(day.water);
        });
    }
    else if (mode === "month") {
        const monthlyMap = new Map();
        
        historyData.forEach(d => {
            const date = new Date(d.time);
            const monthKey = `${date.getMonth()+1}/${date.getFullYear()}`;
            if (!monthlyMap.has(monthKey) || d.time > monthlyMap.get(monthKey).time) {
                monthlyMap.set(monthKey, {
                    temp: d.temp,
                    hum: d.hum,
                    soil: d.soil,
                    water: d.water,
                    time: d.time,
                    label: monthKey
                });
            }
        });
        
        const last6Months = Array.from(monthlyMap.values()).slice(-6);
        last6Months.forEach(month => {
            chart.data.labels.push(month.label);
            chart.data.datasets[0].data.push(month.temp);
            chart.data.datasets[1].data.push(month.hum);
            chart.data.datasets[2].data.push(month.soil);
            chart.data.datasets[3].data.push(month.water);
        });
    }
    
    chart.update();
}

function updateHealth(d){
    let score = 100;
    if(d.temp > 40) score -= 25;
    if(d.soil < 25) score -= 35;
    if(d.hum < 25) score -= 20;
    const waterPercent = waterCmToPercent(d.water);
    if(waterPercent < 20) score -= 20;
    let status="Excellent";
    if(score < 70) status="Warning ⚠️";
    if(score < 40) status="Critical 🚨";
    const systemHealth = document.getElementById("systemHealth");
    if(systemHealth) systemHealth.innerText = status + " ("+Math.max(0,score)+"%)";
}

function calcPlantStress(d){
    let stress = 0;
    if(d.temp > 35) stress += 30;
    if(d.soil < 30) stress += 45;
    if(d.hum < 30) stress += 25;
    let level = "Low 🌿";
    if(stress > 40) level = "Medium ⚠️";
    if(stress > 70) level = "High 🚨";
    const plantStress = document.getElementById("plantStress");
    if(plantStress) plantStress.innerText = level + " (" + stress + "%)";
}

function trackConsumption(percent){
    if(lastWaterLevel !== null){
        let diff = lastWaterLevel - percent;
        if(diff > 0) consumptionHistory.push(diff);
        let total = consumptionHistory.reduce((a,b)=>a+b,0);
        const waterUsage = document.getElementById("waterUsage");
        if(waterUsage) waterUsage.innerText = total.toFixed(2) + "% used";
    }
    lastWaterLevel = percent;
}



function handleAlerts(d){
    const waterPercent = waterCmToPercent(d.water);
    createAlert("soil", d.soil < 25, "🌱 Soil Dry - Needs watering");
    createAlert("temp", d.temp > 42, "🔥 High Temperature");
    createAlert("hum", d.hum < 25, "💨 Low Humidity");
    createAlert("water", waterPercent < 20, "💧 Tank Level Low");
}

function createAlert(id, cond, msg){
    let el = document.getElementById("alert-" + id);
    if(cond){
        if(!el){
            let div = document.createElement("div");
            div.id = "alert-" + id;
            div.className = "alert alert-danger d-flex justify-content-between align-items-center";
            div.innerHTML = `${msg} <button class="btn btn-sm btn-light">OK</button>`;
            if(!playedAlerts.has(id)){
                sound.currentTime = 0;
                sound.play().catch(()=>{});
                playedAlerts.add(id);
            }
            div.querySelector("button").onclick = () => div.remove();
            alertsBox.prepend(div);
        }
    }else{
        if(el){
            el.remove();
            playedAlerts.delete(id);
            let success = document.createElement("div");
            success.className = "alert alert-success";
            success.innerText = `✅ ${msg} resolved`;
            alertsBox.prepend(success);
            setTimeout(()=>{ success.remove(); }, 3000);
        }
    }
}

function showToast(message, type){
    const toast = document.createElement("div");
    toast.className = `toast align-items-center text-white bg-${type === 'success' ? 'success' : 'info'} position-fixed bottom-0 end-0 m-3`;
    toast.setAttribute("role", "alert");
    toast.innerHTML = `<div class="d-flex"><div class="toast-body">${message}</div><button type="button" class="btn-close btn-close-white me-2 m-auto" data-bs-dismiss="toast"></button></div>`;
    document.body.appendChild(toast);
    const bsToast = new bootstrap.Toast(toast);
    bsToast.show();
    toast.addEventListener('hidden.bs.toast', () => toast.remove());
}

            .sort((a, b) => a.sortKey - b.sortKey)
            .slice(-7);
    } 
    else if (reportType === "month") {
        // Monthly: group by month (matching chart's month logic)
        const monthlyMap = new Map();
        
        historyData.forEach(d => {
            const date = new Date(d.time);
            const monthKey = `${date.getMonth()+1}/${date.getFullYear()}`;
            if (!monthlyMap.has(monthKey)) {
                monthlyMap.set(monthKey, { temp: [], hum: [], soil: [], water: [], timestamp: d.time });
            }
            const group = monthlyMap.get(monthKey);
            group.temp.push(d.temp);
            group.hum.push(d.hum);
            group.soil.push(d.soil);
            group.water.push(d.water);
        });
        
        displayData = Array.from(monthlyMap.entries())
            .map(([label, vals]) => ({
                label: label,
                temp: vals.temp.reduce((a,b) => a + b, 0) / vals.temp.length,
                hum: vals.hum.reduce((a,b) => a + b, 0) / vals.hum.length,
                soil: vals.soil.reduce((a,b) => a + b, 0) / vals.soil.length,
                water: vals.water.reduce((a,b) => a + b, 0) / vals.water.length,
                sortKey: vals.timestamp
            }))
            .sort((a, b) => a.sortKey - b.sortKey)
            .slice(-6);
    }
    
    // Display data in table
    if (displayData.length === 0) {
        doc.setTextColor(150, 150, 150);
        doc.text("No historical data available yet. Data will appear as sensors record readings.", 22, y + 5);
        y += 10;
    } else {
        for (let i = 0; i < displayData.length; i++) {
            const item = displayData[i];
            
            if (y > 270) {
                doc.addPage();
                y = 20;
                doc.setFontSize(14);
                doc.setTextColor(...dark);
                doc.text("Sensor Data Analysis (continued)", 20, y);
                y += 8;
                doc.setFillColor(...primary);
                doc.rect(20, y, 170, 8, "F");
                doc.setTextColor(255, 255, 255);
                doc.setFontSize(9);
                doc.text("Time Period", 22, y + 5);
                doc.text("Temp (°C)", 72, y + 5);
                doc.text("Humidity (%)", 108, y + 5);
                doc.text("Soil (%)", 142, y + 5);
                doc.text("Water (%)", 172, y + 5);
                y += 8;
                doc.setTextColor(0, 0, 0);
                doc.setFontSize(8);
            }
            
            // Alternate row background
            if (i % 2 === 0) {
                doc.setFillColor(...lightGray);
                doc.rect(20, y, 170, 7, "F");
            }
            doc.setDrawColor(200);
            doc.rect(20, y, 170, 7);
            
            const tempVal = item.temp.toFixed(1);
            const humVal = item.hum.toFixed(1);
            const soilVal = item.soil.toFixed(1);
            const waterVal = item.water.toFixed(1);
            
            doc.text(String(item.label), 22, y + 5);
            doc.text(tempVal, 74, y + 5);
            doc.text(humVal, 110, y + 5);
            doc.text(soilVal, 144, y + 5);
            doc.text(waterVal, 174, y + 5);
            
            y += 7;
        }
    }
    
    // Add Chart - Capture the existing chart from the page
    doc.addPage();
    doc.setFontSize(14);
    doc.setTextColor(...dark);
    doc.text(`${reportType.toUpperCase()} Analytics Chart`, 20, 20);
    
    // Try to capture the existing chart canvas
    const canvas = document.getElementById("analyticsChart");
    if (canvas) {
        try {
            // Temporary switch chart mode to match report type
            const originalMode = mode;
            mode = reportType;
            buildChart(); // Refresh chart with correct mode
            await new Promise(r => setTimeout(r, 500)); // Wait for chart to update
            
            const imgData = canvas.toDataURL("image/png", 1.0);
            doc.addImage(imgData, "PNG", 15, 35, 180, 100);
            
            // Restore original mode
            mode = originalMode;
            buildChart();
        } catch(e) {
            console.warn("Chart capture failed", e);
            doc.setFontSize(10);
            doc.setTextColor(150, 150, 150);
            doc.text("Chart could not be captured", 20, 50);
        }
    } else {
        doc.setFontSize(10);
        doc.setTextColor(150, 150, 150);
        doc.text("No chart available", 20, 50);
    }
    
    // Add legend
    const legendY = 150;
    const legendItems = [
        { label: "Temperature", color: [255, 99, 132] },
        { label: "Humidity", color: [54, 162, 235] },
        { label: "Soil Moisture", color: [75, 192, 192] },
        { label: "Water Level", color: [255, 206, 86] }
    ];
    
    legendItems.forEach((item, idx) => {
        const x = 20 + (idx * 45);
        doc.setFillColor(item.color[0], item.color[1], item.color[2]);
        doc.rect(x, legendY, 8, 8, "F");
        doc.setFontSize(7);
        doc.setTextColor(0, 0, 0);
        doc.text(item.label, x + 10, legendY + 5);
    });
    
    // Summary Insights
    if (displayData.length > 0) {
        const lastItem = displayData[displayData.length - 1];
        const firstItem = displayData[0];
        
        doc.setFontSize(11);
        doc.setTextColor(...dark);
        doc.text("Key Insights:", 20, 170);
        doc.setFontSize(9);
        doc.setTextColor(0, 0, 0);
        
        const tempChange = (lastItem.temp - firstItem.temp).toFixed(1);
        const soilChange = (lastItem.soil - firstItem.soil).toFixed(1);
        
        doc.text(`• Temperature trend: ${tempChange > 0 ? '+' : ''}${tempChange}°C`, 20, 180);
        doc.text(`• Soil moisture trend: ${soilChange > 0 ? '+' : ''}${soilChange}%`, 20, 188);
        doc.text(`• Average humidity: ${(displayData.reduce((a,b) => a + b.hum, 0) / displayData.length).toFixed(1)}%`, 20, 196);
        
        // Recommendation
        let recommendation = "";
        const avgSoil = displayData.reduce((a,b) => a + b.soil, 0) / displayData.length;
        if (avgSoil < 30) {
            recommendation = "⚠️ Recommendation: Soil is dry. Consider increasing irrigation.";
        } else if (avgSoil > 70) {
            recommendation = "✅ Recommendation: Soil moisture is good. Reduce irrigation to prevent overwatering.";
        } else {
            recommendation = "🌿 Recommendation: Soil conditions are optimal. Maintain current schedule.";
        }
        doc.text(recommendation, 20, 208);
    }
    
    // Footer
    doc.setFillColor(...primary);
    doc.rect(0, 285, 210, 12, "F");
    doc.setTextColor(255, 255, 255);
    doc.setFontSize(9);
    doc.text("HydroGen AI System © 2026", 20, 292);
    doc.text("Smart Irrigation • Data Driven • Sustainable", 105, 292);
    
    // Save PDF
    doc.save(`HydroGen_${reportType}_Report.pdf`);
};

const WEATHER_API_KEY = "5dd74768dc40a34a27ac51503c655bec";
const CITY = "Port Said";

window.loadWeather = async function() {
    const weatherMain = document.getElementById("weatherMain");
    const weatherForecast = document.getElementById("weatherForecast");
    const refreshBtn = document.getElementById("weatherRefreshBtn");
    
    if (!weatherMain || !weatherForecast) return;
    
    if (refreshBtn) {
        refreshBtn.disabled = true;
        refreshBtn.innerHTML = '<i class="fas fa-spinner fa-spin"></i> Refresh';
    }
    
    try {
        const res = await fetch(`https://api.openweathermap.org/data/2.5/forecast?q=${CITY}&appid=${WEATHER_API_KEY}&units=metric`);
        const data = await res.json();
        
        if (data.cod !== "200") throw new Error();
        
        const now = new Date();
        weatherMain.innerHTML = `
            <div class="d-flex justify-content-between align-items-center">
                <div>
                    <h5 class="mb-0">📍 ${CITY}</h5>
                    <small>${now.toLocaleDateString("en-US", { weekday: "long", year: "numeric", month: "long", day: "numeric" })}</small>
                </div>
                <div class="text-end">
                    <div class="display-6">${Math.round(data.list[0].main.temp)}°C</div>
                    <small>${data.list[0].weather[0].description}</small>
                </div>
            </div>
        `;
        
        const rainExpected = data.list.some(i => i.weather[0].main.toLowerCase().includes("rain"));
        weatherRainExpected = rainExpected;
        
        let forecastHtml = '';
        const next6 = data.list.slice(0, 6);
        next6.forEach(item => {
            const time = new Date(item.dt_txt);
            forecastHtml += `
                <div class="forecast-item">
                    <div class="small">${time.toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' })}</div>
                    <img src="https://openweathermap.org/img/wn/${item.weather[0].icon}.png" width="35">
                    <div class="fw-bold">${Math.round(item.main.temp)}°C</div>
                    <small class="d-block">${item.weather[0].description}</small>
                </div>
            `;
        });
        weatherForecast.innerHTML = forecastHtml;
        
    } catch (err) {
        console.error(err);
        weatherMain.innerHTML = "❌ Weather connection error";
    } finally {
        if (refreshBtn) {
            refreshBtn.disabled = false;
            refreshBtn.innerHTML = '<i class="fas fa-sync-alt"></i> Refresh';
        }
    }
};

const AI_API = "https://openrouter.ai/api/v1/chat/completions";

async function sendMessage(){
    const userInput = document.getElementById("userInput");
    const msg = userInput.value.trim();
    if(!msg) return;
    addMessage("You", msg);
    userInput.value = "";
    addMessage("AI", "Typing...", "typing");
    try{
        const res = await fetch(AI_API,{
            method:"POST",
            headers:{
                "Authorization": `Bearer ${AI_KEY}`,
                "Content-Type":"application/json"
            },
            body: JSON.stringify({
                model:"openai/gpt-3.5-turbo",
                messages:[
                    {role:"system", content: getSystemContext()},
                    {role:"user", content: msg}
                ]
            })
        });
        const data = await res.json();
        removeTyping();
        const reply = data.choices?.[0]?.message?.content || "AI error";
        addMessage("AI", reply);
    }catch(e){
        removeTyping();
        addMessage("AI","❌ AI connection failed");
    }
}

function getSystemContext(){
    const temp = envTemp.innerText;
    const hum = envHum.innerText;
    const soil = envSoil.innerText;
    const pumpState = pumpText.innerText;
    const aiState = isAIActive ? "ON" : "OFF";
    const rainExpected = weatherRainExpected ? "Yes" : "No";
    return `You are HydroGen AI assistant for smart irrigation. 
Current readings: Temperature: ${temp}°C, Humidity: ${hum}%, Soil Moisture: ${soil}%, 
Pump: ${pumpState}, AI Mode: ${aiState}, Rain Expected: ${rainExpected}.
Provide short, helpful advice about irrigation, plant care, and water conservation. Be friendly and professional.`;
}

function addMessage(sender, text, type="normal"){
    const chatBox = document.getElementById("chatBox");
    if(!chatBox) return;
    
    if (type !== "typing") {
        removeTyping();
    }
    
    const div = document.createElement("div");
    div.className = `msg ${sender.toLowerCase()} ${type}`;
    
    if (sender.toLowerCase() === "ai") {
        div.innerHTML = `<span><b>🤖 HydroGen AI:</b><br>${text.replace(/\n/g, '<br>')}</span>`;
    } else {
        div.innerHTML = `<span><b>👤 You:</b><br>${text.replace(/\n/g, '<br>')}</span>`;
    }
    
    chatBox.appendChild(div);
    chatBox.scrollTop = chatBox.scrollHeight;
}

function removeTyping(){
    const typingElements = document.querySelectorAll(".typing");
    typingElements.forEach(e => e.remove());
}

function quickAsk(question){
    const userInput = document.getElementById("userInput");

        userInput.value = question;
        sendMessage();
    }
}

        w.style.display = isVisible ? "none" : "flex";
        if (!isVisible && !chatInitialized) {
            initAIChat();
        }
    }
}
